-- 메뉴테이블
create table menu_yp(
    menu_id VARCHAR(10) PRIMARY KEY,
    menu_name VARCHAR(100) NOT NULL,
    menu_price INT,
    menu_category VARCHAR(100) CHECK
    (menu_category IN ('커피','논-커피','에이드/차','스무디','아이스 블렌딩','버블티','요거트','디저트'))
);
-- 고객테이블
CREATE TABLE customer_yp (
    customer_id VARCHAR(50) PRIMARY KEY,
    customer_name VARCHAR(100) NOT NULL,
    phone_number VARCHAR(15) CHECK (phone_number LIKE '010-____-____') NOT NULL,
    membership_grade VARCHAR(10) CHECK (membership_grade IN ('브론즈', '실버', '골드'))
);

create table order_yp(
    order_id VARCHAR(50) PRIMARY KEY, --주문id
    customer_id VARCHAR(50), --고객id
    menu_id VARCHAR(10), -- 메뉴id
    menu_number int not null, -- 주문 수량
    total_amount int not null, -- 총 주문금액
    order_date date not null, -- 주문날짜
    FOREIGN KEY (customer_id) references customer_yp(customer_id), --외래키
    FOREIGN KEY (menu_id) references menu_yp(menu_id)  --외래키
    );

select * from order_yp;

select *
from customer_yp c
join menu_yp m
on m.menu_id = c.customer_id
join order_yp o
on o.order_id = m.menu_id;
